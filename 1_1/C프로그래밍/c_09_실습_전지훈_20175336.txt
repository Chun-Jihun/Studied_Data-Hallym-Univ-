#include <stdio.h>
#include <conio.h>

int main(void){
    char str[2][100] = {0,};
    int check;

    printf("c_09_전지훈_20175336\n");

    do{
        int i = 0;
        printf("문자열 입력 : ");
        gets(str[0]);
        if(str[0][0] == '\0'){
            printf("<Enter키>");
            break;
        }
        printf("문자열 입력 : ");
        gets(str[1]);

        while(str[0][i]){
            if(str[0][i+1] != str[1][i+1]){
                check = 0;
                break;
            }
            else {
                check = 1;
            }
            i++;
        }
        check ? printf("%s, %s, 같음\n", str[0], str[1]) : printf("%s, %s, 다름\n", str[0], str[1]);
    }while(1);

    return 0;
}
